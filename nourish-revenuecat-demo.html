Implementation
Core RevenueCat service, subscription state store, and entitlement checking logic for the Base44 React wrapper.
services/revenuecatService.ts
import Purchases, { PurchasesPackage } from 'react-native-purchases';

// Initialize RC SDK — call from App.tsx on mount
export const initRevenueCat = () => {
Purchases.configure({ apiKey: RC_IOS_API_KEY });
};

// Fetch subscription offerings from RC dashboard
export const getOfferings = async () => {
const offerings = await Purchases.getOfferings();
return offerings.current?.availablePackages ?? [];
};

// Purchase a package (monthly or annual)
export const purchasePackage = async (pkg: PurchasesPackage) => {
const { customerInfo } = await Purchases.purchasePackage(pkg);
return hasEntitlement(customerInfo);
};

// Check premium_access entitlement
export const hasEntitlement = (info: any) =>
!!info?.entitlements?.active?.['premium_access'];
store/subscriptionStore.ts
import { create } from 'zustand';

interface SubscriptionState {
isSubscribed: boolean;
planId: 'monthly' | 'annual' | null;
trialActive: boolean;
trialDaysLeft: number;
entitlementActive: boolean;
}

export const useSubscriptionStore = create<SubscriptionState>((set) => ({
isSubscribed: false,
planId: null,
trialActive: false,
trialDaysLeft: 7,
entitlementActive: false,
setSubscribed: (plan, trial) =>
set({ isSubscribed: true, planId: plan,
trialActive: trial, entitlementActive: true }),
}));
